SET(GUIDESRC ${CMAKE_CURRENT_SOURCE_DIR})
SET(GUIDE ${CMAKE_CURRENT_BINARY_DIR})

FIND_PROGRAM(PDFLATEX pdflatex)
MARK_AS_ADVANCED(PDFLATEX)
FIND_PROGRAM(BIBTEX bibtex)
MARK_AS_ADVANCED(BIBTEX)
FIND_PROGRAM(MAKEINDEX makeindex)
MARK_AS_ADVANCED(MAKEINDEX)
FIND_PROGRAM(HTLATEX htlatex)
MARK_AS_ADVANCED(HTLATEX)


ADD_CUSTOM_TARGET(document-pdf
	COMMAND ${PDFLATEX} --output-directory ${GUIDE} ${GUIDESRC}/document.tex
    COMMAND BIBINPUTS=${GUIDESRC} BSTINPUT=${GUIDE} TEXMFOUTPUT=${GUIDE} ${BIBTEX} ${GUIDE}/document
	COMMAND ${PDFLATEX} --output-directory ${GUIDE} ${GUIDESRC}/document.tex
	COMMAND ${PDFLATEX} --output-directory ${GUIDE} ${GUIDESRC}/document.tex
	WORKING_DIRECTORY ${GUIDESRC}
)
